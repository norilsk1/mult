void printGeerM (geer& tempgeer)
{
  cout<<tempgeer.stup<<" колесo: "<<endl;
    cout<<"  Z"<<tempgeer.stup<<"-"<<tempgeer.z<<"  ";
    cout<<"  M-"<<tempgeer.m<<"   ";
    cout<<"  D_d-"<<tempgeer.D_delit<<"   ";
    cout<<"  D_N-"<<tempgeer.D_N<<"   ";
    cout<<"  N об/мин-"<<tempgeer.n_ob<<endl;
}

void calc(geer& g1, geer& g2, double Npp)
{
      g1.D_delit=228.6/(1+1/Npp);
//        temp=geer1.D_delit;
        g1.calcAll();
        cout<<"смещение" <<g1.stup<<"-го колеса: "<<g1.delta<<endl;
        g2.D_delit=228.6-g1.D_delit;

        g2.calcAll();
        cout<<"смещение " <<g2.stup<<"-го колеса: "<<g2.delta<<endl;
        g2.calcOb(g1);
}
